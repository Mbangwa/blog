1.Write a program to manage withdrawals from a bank account. If there are enough funds, the program should accept the withdrawal, otherwise it should reject it.

import java.util.*;
/**
* @author Mbangwa (inno).
*
*/
class Account{
private double balance;

//constructor
public Account(double balance) {
setBalance(balance);
}

//Getters and setters
public void setBalance(double balance) {
this.balance = balance;
}

public double getBalance() {
return balance;
}

//withdraw method
public void withdraw(int withdraw_amount) {
//tucheki kama kuna balance yyte
if(balance != 0 && withdraw_amount < balance) {
balance -= withdraw_amount;
System.out.println("Umefanikiwa kutoa Sh. " + withdraw_amount);
System.out.println("Salio: " + balance);
}else{
//kama ela amna uezi toa chochote
System.out.println("Withdraw rejected, insufficient fund");
}
}
}

public class Main {
public static void main(String[] args) {
Account account = new Account(10000); // tumeanzisha akaunt kwa sh 10000
Scanner scan = new Scanner(System.in);
System.out.print("Ingiza kiasi: "):
int a = scan.nextInt();
account.withdraw(a);

}
}

Output:

Ingiza kiasi: 2000
Umefanikiwa kutoa Sh. 2000
Salio: 8000.0

5.Write a program that asks user to enter marks and the program will output grade according to the following table

Marks 0-34 35-39 40-49 50-59 60-69 70-100
Grade E D C B B+ A

Answer:

import java.util.Scanner;

//written by Mbangwa (inno).

public class Main {
public static void main(String [] args) {
Scanner sc = new Scanner(System.in);
System.out.print("Enter your marks: ");
int marks = sc.nextInt();

if(marks >= 70 && marks <= 100) {
System.out.println("Grade: A");
}
else if(marks >= 60 && marks <= 69) {
System.out.println("Grade: B+");
}
else if(marks >= 50 && marks <= 59) {
System.out.println("Grade: B");
}
else if(marks >= 40 && marks <= 49) {
System.out.println("Grade: C");
}
else if(marks >= 35 && marks <= 39) {
System.out.println("Grade: D");
}
else if(marks >= 0 && marks <= 34) {
System.out.println("Grade: E");
}
else {
System.out.println("Invalid marks");
}
}
}

6.Write the program that computes net payment of an employee. The program will receive the hours worked and payment rate per hour. The employee is taxed 20% of his gross income if it exceeds 100,000/=

Answer:

import java.util.*;

// written by mbangwa (inno)
class Gross{
private double gross;
private int hours;

//constructor
public Gross(double gross, int hours) {
setGross(gross * hours);
setHours(hours);
}

//setter and getters
public void setGross(double gross) {
this.gross = gross;
}

public void setHours(int hours) {
this.hours = hours;
}

public double getGross() {
return gross;
}

/*Note, sijaeka getters za hours na rate bcoz cna haja nazo kweny output, only gross ndo tuna haja nayo*/

public void calculateGross() {
if(gross > 100000) {
double less = gross * 0.2;
gross -= less;
System.out.print("Gross: " + this.gross);
}
else {
System.out.print("Gross: " + this.gross);
}
}
}

public class Main {

public static void main(String[] args) {
Scanner scan = new Scanner(System.in);
System.out.print("Deposit initial money: ");
int g = scan.nextInt();
System.out.print("Enter hours of work: ");
int h = scan.nextInt();

Gross gross = new Gross(g, h);
gross.calculateGross();

}
}

7.Write a program to accept characters as input, and quit when a 'Q‘ or a 'q‘ is typed, printing how many characters other than 'q‘ or 'Q‘ have been typed.

Answer:

import java.util.*;
/**
* @author Mbangwa (inno).
*
*/
public class Main {
public static void main(String[] args) {

Scanner scan = new Scanner(System.in);
char ch;
int count = 0;
do {
System.out.print(">> ");
ch = scan.next().charAt(0);
count++;
} while(ch != 'q' && ch != 'Q');

System.out.println("You typed " + (count-1) + " characters");
}
}
